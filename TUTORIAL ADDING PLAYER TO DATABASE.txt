----------------SETUP ARDUINO-------------------

1. UPLOAD THIS INO FILE BELOW TO ARDUINO MEGA

#include <SPI.h>
#include <MFRC522.h>

#define SS_PIN 53    // SDA -> Pin 53 on Arduino Mega
#define RST_PIN 5    // RST -> Any digital pin, here we use 5

MFRC522 mfrc522(SS_PIN, RST_PIN);

void setup() {
  Serial.begin(9600);
  SPI.begin();
  mfrc522.PCD_Init();
}

void loop() {
  if (!mfrc522.PICC_IsNewCardPresent()) return;
  if (!mfrc522.PICC_ReadCardSerial()) return;

  // Construct UID string
  String uidStr = "";
  for (byte i = 0; i < mfrc522.uid.size; i++) {
    if (mfrc522.uid.uidByte[i] < 0x10) uidStr += "0"; // Pad with 0 if needed
    uidStr += String(mfrc522.uid.uidByte[i], HEX);
  }

  uidStr.toUpperCase();        // Make it uppercase (Python expects this)
  Serial.println(uidStr);      // Send UID only, no extra text

  delay(1000); // Simple debounce
}

2. CONNECT ARDUINO MEGA TO PC AND NOTE THE PORT DIRECTORY (COM1,COM2,dev/usbttys0,etc.)
